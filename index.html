Função para carregar conteúdo baseado na URL function loadFromURL() { let path = window.location.pathname; path = path.replace(/^\/|\/$/g, ''); // remove barras no início/fim // Remove possíveis extensões ou parâmetros path = path.split('?')[0].split('.html')[0]; const storyId = urlToIdMap[path]; if (storyId && typeof loadStory === 'function') { loadStory(storyId); } else if (path === '' || path === 'index.html') { // Página inicial loadStory('ze-pilintra'); } } // 3. Função para navegar sem recarregar página, atualizando URL function navigateTo(storyId, urlPath) { if (typeof loadStory === 'function') { loadStory(storyId); } history.pushState(null, '', urlPath); document.title = (bancoDados[storyId]?.titulo || 'Seu Zé Pilintra') + ' | Portal da Malandragem'; if (typeof closeMenu === 'function') closeMenu(); } // 4. Modificar todos os links com onclick para usarem navigateTo function upgradeLinksToSeo() { // Links dos Exus e Pombagiras gerados dinamicamente (já estão com addEventListener) // Vamos capturar os links estáticos do menu (Pretos-Velhos, Caboclos, etc.) const allLinks = document.querySelectorAll('#sidebar a[onclick*="loadStory"]'); allLinks.forEach(link => { const onclickAttr = link.getAttribute('onclick'); const match = onclickAttr.match(/loadStory\(['"]([^'"]+)['"]/); if (match) { const storyId = match[1]; // Encontrar a URL amigável correspondente let urlPath = ''; for (let [path, id] of Object.entries(urlToIdMap)) { if (id === storyId) { urlPath = '/' + path; break; } } if (urlPath) { link.setAttribute('href', urlPath); link.removeAttribute('onclick'); // clonar para remover event listeners antigos const newLink = link.cloneNode(true); link.parentNode.replaceChild(newLink, link); newLink.addEventListener('click', (e) => { e.preventDefault(); navigateTo(storyId, urlPath); }); } } }); // Tratar também os links dinâmicos do menu de Exus/Pombagiras (criados via JS) const dynamicExus = document.querySelectorAll('#exusSubmenu a, #pomboSubmenu a'); dynamicExus.forEach(link => { const entity = link.getAttribute('data-entity'); if (entity) { let urlPath = ''; for (let [path, id] of Object.entries(urlToIdMap)) { if (id === entity) { urlPath = '/' + path; break; } } if (urlPath) { link.setAttribute('href', urlPath); link.removeEventListener('click', link._listener); const newLink = link.cloneNode(true); link.parentNode.replaceChild(newLink, link); newLink.addEventListener('click', (e) => { e.preventDefault(); navigateTo(entity, urlPath); }); } } }); } // 5.
